CREATE TABLE PCI_DUMMY(
	COUNTY VARCHAR,
	STATE VARCHAR,
	PER_CAPITA_INCOME VARCHAR,
	MEDIAN_HOUSEHOLD_INCOME VARCHAR,
	MEDIAN_FAMILY_INCOME VARCHAR,
	POPULATION VARCHAR,
	NUMBER_OF_HOUSEHOLDS VARCHAR
)

CREATE TABLE PCI(
	KEY INTEGER PRIMARY KEY AUTOINCREMENT,
	COUNTY INTEGER,
	STATE INTEGER,
	PER_CAPITA_INCOME INTEGER,
	MEDIAN_HOUSEHOLD_INCOME INTEGER,
	MEDIAN_FAMILY_INCOME INTEGER,
	POPULATION INTEGER,
	NUMBER_OF_HOUSEHOLDS INTEGER
)



INSERT INTO PCI(COUNTY,STATE,PER_CAPITA_INCOME,MEDIAN_HOUSEHOLD_INCOME,MEDIAN_FAMILY_INCOME,POPULATION,NUMBER_OF_HOUSEHOLDS)
SELECT COUNTY_CODE,STATE_CODE,PER_CAPITA_INCOME,MEDIAN_HOUSEHOLD_INCOME,MEDIAN_FAMILY_INCOME,POPULATION,NUMBER_OF_HOUSEHOLDS FROM PCI_DUMMY, SC_CODES WHERE PCI_DUMMY.STATE = SC_CODES.STATE_NAME AND PCI_DUMMY.COUNTY = SC_CODES."CITY_NAME/COUNTY_NAME" AND SC_CODES.CITY_CODE IS NULL;
